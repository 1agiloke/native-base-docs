  render() {
    return (
      &lt;Container>
        &lt;Header />
        &lt;Content>
          &lt;ListItem selected={false} >
            &lt;Left>
              &lt;Text>Lunch Break&lt;/Text>
            &lt;/Left>
            &lt;Right>
              &lt;Radio
                color={"#f0ad4e"}
                selectedColor={"#5cb85c"}
                selected={false}
              />
            &lt;/Right>
          &lt;/ListItem>
          &lt;ListItem selected={true}>
            &lt;Left>
              &lt;Text>Discussion with Client&lt;/Text>
            &lt;/Left>
            &lt;Right>
              &lt;Radio
                color={"#f0ad4e"}
                selectedColor={"#5cb85c"}
                selected={true}
              />
            &lt;/Right>
          &lt;/ListItem>
        &lt;/Content>
      &lt;/Container>
    );
  }
}</code></pre><br />
